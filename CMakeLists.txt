cmake_minimum_required(VERSION 3.0)
project(guji)
set(SRC_LIST main.c engine/source/runtime/engine.c engine/source/runtime/function/render/window_system.c)
add_executable(${PROJECT_NAME} ${SRC_LIST})
target_include_directories(${PROJECT_NAME} PUBLIC engine/source/runtime)
set(GLFW_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
add_subdirectory(engine/dependencies/glfw)
target_link_libraries(guji glfw)
